\newcommand{\HeadingItemPlain}[2]{
	\item
		\begin{tabular*}{0.97\textwidth}[t]{l@{\extracolsep{\fill}}r}
			\textbf{#1} & {#2}
		\end{tabular*} \\
}

\newcommand{\Text}[1]{
	\begin{minipage}{0.97\textwidth}
		#1
	\end{minipage}\vspace{-5pt}
}

\newcommand{\SubHeadingItem}[2]{
	\begin{tabular*}{0.97\textwidth}{l@{\extracolsep{\fill}}r}
		\textit{\small#1} & \textit{\small #2}
	\end{tabular*}
}

\newcommand{\HeadingItem}[4]{
	\HeadingItemPlain{#1}{#2}
	\SubHeadingItem{#3}{#4}
}

\newcommand{\Item}[2]{
	\item\small{
		\textbf{#1}{: #2}
	}
	\vspace{-2pt} % -2pt or -4pt
}

\renewcommand{\labelitemii}{$\circ$}

\newcommand{\HeadingItemListStart}{\begin{itemize}[leftmargin=*]}
\newcommand{\HeadingItemListEnd}{\end{itemize}}

\newcommand{\ItemListStart}{\begin{itemize}\vspace{-5pt}}
\newcommand{\ItemListEnd}{\end{itemize}\vspace{-5pt}}
